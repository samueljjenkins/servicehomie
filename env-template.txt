# Whop App Environment Variables
# Copy this content to a new file called .env.local and replace the placeholder values with your actual values from the Whop dashboard

# Your Whop API Key (get this from the Whop dashboard under apps)
WHOP_API_KEY="your_whop_api_key_here"

# Webhook secret (get this after creating a webhook in the app settings screen)
WHOP_WEBHOOK_SECRET="your_webhook_secret_here"

# Agent User ID (use any Whop user ID your app can control)
NEXT_PUBLIC_WHOP_AGENT_USER_ID="your_agent_user_id_here"

# App ID (use the corresponding app ID to the secret API key)
NEXT_PUBLIC_WHOP_APP_ID="your_app_id_here"

# Company ID (use the corresponding company ID to the secret API key)
NEXT_PUBLIC_WHOP_COMPANY_ID="your_company_id_here"
